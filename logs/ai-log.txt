// ai/utils.js (add log function)
const fs = require('fs');
const path = require('path');

function writeLog(message) {
    const logPath = path.join(__dirname, '../logs/ai-log.txt');
    const timestamp = new Date().toISOString();
    fs.appendFileSync(logPath, `[${timestamp}] ${message}\n`);
}

module.exports = { fetchAllApps, updateAppData, writeLog };
